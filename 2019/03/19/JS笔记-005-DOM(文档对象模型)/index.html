    }
else
    {
    alert("未启用 cookie")
    }
}
</script>

onmouseover 和 onmouseout 事件

onmouseover 和 onmouseout 事件可用于在用户的鼠标移至 HTML 元素上方或移出元素时触发函数。

<!DOCTYPE html>
<html>
<body>

<div onmouseover="mOver(this)" onmouseout="mOut(this)" style="background-color:green;width:120px;height:20px;padding:40px;color:#ffffff;">把鼠标移到上面</div>

<script>
function mOver(obj)
{
obj.innerHTML="谢谢"
}

function mOut(obj)
{
obj.innerHTML="把鼠标移到上面"
}
</script>

</body>
</html>
<!DOCTYPE html>
<html>
<body>

<h1 onmouseover="style.color='red'" onmouseout="style.color='blue'">
请把鼠标移到这段文本上
</h1>

</body>
</html>
onmousedown、onmouseup 以及 onclick 事件

onmousedown, onmouseup 以及 onclick 构成了鼠标点击事件的所有部分。首先当点击鼠标按钮时，会触发 onmousedown 事件，当释放鼠标按钮时，会触发 onmouseup 事件，最后，当完成鼠标点击时，会触发 onclick 事件。

<!DOCTYPE html>
<html>
<body>

<div onmousedown="mDown(this)" onmouseup="mUp(this)" onclick = alert(6) style="background-color:green;color:#ffffff;width:90px;height:20px;padding:40px;font-size:12px;">请点击这里</div>

<script>
function mDown(obj)
{
obj.style.backgroundColor="#1ec5e5";
obj.innerHTML="请释放鼠标按钮"
}

function mUp(obj)
{
obj.style.backgroundColor="green";
obj.innerHTML="请按下鼠标按钮"
}
</script>

</body>
</html>
添加新的HTML元素

如需向 HTML DOM 添加新元素，您必须首先创建该元素（元素节点），然后向一个已存在的元素追加该元素。

<!DOCTYPE html>
<html>
<body>

<div id="div1">
<p id="p1">这是一个段落。</p>
<p id="p2">这是另一个段落。</p>
</div>
<div id="div2">
<p id="p3">这是asdasdwqeqwasd段落。</p>
</div>

<script>
//创建新的标签
var para=document.createElement("p");
//创建一个新的文本节点
var node=document.createTextNode("这是新段落。");
//想标签追加该节点
para.appendChild(node);

//找到一个新的元素
var element=document.getElementById("div1");
///追加新的标签
element.appendChild(para);
document.getElementById("div2").appendChild(para);
</script>

</body>
</html>
删除已有的HTML元素

如需删除 HTML 元素，您必须首先获得该元素的父元素：

<!DOCTYPE html>
<html>
<body>

<div id="div1">
<p id="p1">这是一个段落。</p>
<p id="p2">这是另一个段落。</p>
</div>

<script>
//先获得父节点的对象
var parent=document.getElementById("div1");
//获得要删除的那个控件对象
var child=document.getElementById("p1");
//通过父元素删除字元素
parent.removeChild(child);
</script>

</body>
</html>

或者直接获得子元素，通过parentNode获得父元素来删除

var child=document.getElementById("p1");
child.parentNode.removeChild(child);
